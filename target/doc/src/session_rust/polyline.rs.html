191    pub fn shift(&mut self, times: i32) {
192        if self.points.is_empty() {
193            return;
194        }
195        let len = self.points.len();
196        let shift_amount = ((times % len as i32) + len as i32) % len as i32;
197        self.points.rotate_left(shift_amount as usize);
198    }

201    pub fn length_squared(&self) -> f64 {
202        let mut length = 0.0f64;
203        for i in 0..self.segment_count() {
204            let segment = self.points[i + 1].clone() - self.points[i].clone();

210    /// Get point at parameter t along a line segment (t=0 is start, t=1 is end)
211    pub fn point_at_parameter(start: &Point, end: &Point, t: f64) -> Point {
212        let s = 1.0 - t;
213        let t_f32 = t;
214        let s_f32 = s;
215        Point::new(
216            if start.x() == end.x() {
217                start.x()
218            } else {
219                s_f32 * start.x() + t_f32 * end.x()
220            },
221            if start.y() == end.y() {
222                start.y()
223            } else {
224                s_f32 * start.y() + t_f32 * end.y()
225            },
226            if start.z() == end.z() {
227                start.z()
228            } else {
229                s_f32 * start.z() + t_f32 * end.z()
230            },
231        )
232    }

234    /// Find closest point on line segment to given point, returns parameter t
235    pub fn closest_point_to_line(point: &Point, line_start: &Point, line_end: &Point) -> f64 {
236        let d = line_end.clone() - line_start.clone();
237        let dod = d.length_squared();
238
239        if dod > 0.0 {
240            if (point.clone() - line_start.clone()).length_squared()
241                <= (point.clone() - line_end.clone()).length_squared()
242            {
243                (point.clone() - line_start.clone()).dot(&d) / dod
244            } else {
245                1.0 + (point.clone() - line_end.clone()).dot(&d) / dod
246            }
247        } else {
248            0.0
249        }
250    }

252    /// Check if two line segments overlap and return the overlapping segment
253    pub fn line_line_overlap(
254        line0_start: &Point,
255        line0_end: &Point,
256        line1_start: &Point,
257        line1_end: &Point,
258    ) -> Option<(Point, Point)> {
259        let mut t = [0.0, 1.0, 0.0, 0.0];
260        t[2] = Self::closest_point_to_line(line1_start, line0_start, line0_end);
261        t[3] = Self::closest_point_to_line(line1_end, line0_start, line0_end);
262
263        let do_overlap = !((t[2] < 0.0 && t[3] < 0.0) || (t[2] > 1.0 && t[3] > 1.0));
264        t.sort_by(|a, b| a.partial_cmp(b).unwrap());
265
266        let overlap_valid = (t[2] - t[1]).abs() > Tolerance::ZERO_TOLERANCE;
267
268        if do_overlap && overlap_valid {
269            Some((
270                Self::point_at_parameter(line0_start, line0_end, t[1]),
271                Self::point_at_parameter(line0_start, line0_end, t[2]),
272            ))
273        } else {
274            None
275        }
276    }

298    /// Calculate overlap average of two line segments
299    pub fn line_line_overlap_average(
300        line0_start: &Point,
301        line0_end: &Point,
302        line1_start: &Point,
303        line1_end: &Point,
304    ) -> (Point, Point) {
305        let line_a = Self::line_line_overlap(line0_start, line0_end, line1_start, line1_end);
306        let line_b = Self::line_line_overlap(line1_start, line1_end, line0_start, line0_end);
307
308        if let (Some((line_a_start, line_a_end)), Some((line_b_start, line_b_end))) =
309            (line_a, line_b)
310        {
311            let mid_line0_start = Point::new(
312                (line_a_start.x() + line_b_start.x()) * 0.5,
313                (line_a_start.y() + line_b_start.y()) * 0.5,
314                (line_a_start.z() + line_b_start.z()) * 0.5,
315            );
316            let mid_line0_end = Point::new(
317                (line_a_end.x() + line_b_end.x()) * 0.5,
318                (line_a_end.y() + line_b_end.y()) * 0.5,
319                (line_a_end.z() + line_b_end.z()) * 0.5,
320            );
321            let mid_line1_start = Point::new(
322                (line_a_start.x() + line_b_end.x()) * 0.5,
323                (line_a_start.y() + line_b_end.y()) * 0.5,
324                (line_a_start.z() + line_b_end.z()) * 0.5,
325            );
326            let mid_line1_end = Point::new(
327                (line_a_end.x() + line_b_start.x()) * 0.5,
328                (line_a_end.y() + line_b_start.y()) * 0.5,
329                (line_a_end.z() + line_b_start.z()) * 0.5,
330            );
331
332            let mid0_vec = mid_line0_end.clone() - mid_line0_start.clone();
333            let mid1_vec = mid_line1_end.clone() - mid_line1_start.clone();
334
335            if mid0_vec.length_squared() > mid1_vec.length_squared() {
336                (mid_line0_start, mid_line0_end)
337            } else {
338                (mid_line1_start, mid_line1_end)
339            }
340        } else {
341            Self::line_line_average(line0_start, line0_end, line1_start, line1_end)
342        }
343    }

345    /// Create line from projected points onto a base line
346    pub fn line_from_projected_points(
347        line_start: &Point,
348        line_end: &Point,
349        points: &[Point],
350    ) -> Option<(Point, Point)> {
351        if points.is_empty() {
352            return None;
353        }
354
355        let mut t_values: Vec<f64> = points
356            .iter()
357            .map(|p| Self::closest_point_to_line(p, line_start, line_end))
358            .collect();
359
360        t_values.sort_by(|a, b| a.partial_cmp(b).unwrap());
361
362        let output_start = Self::point_at_parameter(line_start, line_end, t_values[0]);
363        let output_end =
364            Self::point_at_parameter(line_start, line_end, t_values[t_values.len() - 1]);
365
366        if (t_values[0] - t_values[t_values.len() - 1]).abs() > Tolerance::ZERO_TOLERANCE {
367            Some((output_start, output_end))
368        } else {
369            None
370        }
371    }

373    /// Find closest distance and point from a point to this polyline
374    pub fn closest_distance_and_point(&self, point: &Point) -> (f64, usize, Point) {
375        let mut edge_id = 0;
376        let mut closest_distance = f64::MAX;
377        let mut best_t = 0.0;
378
379        for i in 0..self.segment_count() {
380            let t = Self::closest_point_to_line(point, &self.points[i], &self.points[i + 1]);
381            let point_on_segment =
382                Self::point_at_parameter(&self.points[i], &self.points[i + 1], t);
383            let distance = point.distance(&point_on_segment);
384
385            if distance < closest_distance {
386                closest_distance = distance;
387                edge_id = i;
388                best_t = t;
389            }
390
391            if closest_distance < Tolerance::ZERO_TOLERANCE {
392                break;
393            }
394        }
395
396        let closest_point =
397            Self::point_at_parameter(&self.points[edge_id], &self.points[edge_id + 1], best_t);
398        (closest_distance, edge_id, closest_point)
399    }

413    /// Calculate center point of polyline
414    pub fn center(&self) -> Point {
415        if self.points.is_empty() {
416            return Point::new(0.0, 0.0, 0.0);
417        }
418
419        let n = if self.is_closed() && self.points.len() > 1 {
420            self.points.len() - 1
421        } else {
422            self.points.len()
423        };
424
425        let mut sum_x = 0.0;
426        let mut sum_y = 0.0;
427        let mut sum_z = 0.0;
428
429        for i in 0..n {
430            sum_x += self.points[i].x();
431            sum_y += self.points[i].y();
432            sum_z += self.points[i].z();
433        }
434
435        Point::new(sum_x / n as f64, sum_y / n as f64, sum_z / n as f64)
436    }

444    /// Get average plane from polyline points
445    pub fn get_average_plane(&self) -> (Point, Vector, Vector, Vector) {
446        let origin = self.center();
447
448        let x_axis = if self.points.len() >= 2 {
449            let mut x = self.points[1].clone() - self.points[0].clone();
450            x.normalize_self();
451            x
452        } else {
453            Vector::new(1.0, 0.0, 0.0)
454        };
455
456        let z_axis = self.average_normal();
457        let mut y_axis = z_axis.cross(&x_axis);
458        y_axis.normalize_self();
459
460        (origin, x_axis, y_axis, z_axis)
461    }

517    /// Extend polyline segment
518    pub fn extend_segment(
519        &mut self,
520        segment_id: usize,
521        dist0: f64,
522        dist1: f64,
523        proportion0: f64,
524        proportion1: f64,
525    ) {
526        if segment_id >= self.segment_count() {
527            return;
528        }
529
530        let mut p0 = self.points[segment_id].clone();
531        let mut p1 = self.points[segment_id + 1].clone();
532        let v = p1.clone() - p0.clone();
533
534        if proportion0 != 0.0 || proportion1 != 0.0 {
535            p0 -= v.clone() * proportion0;
536            p1 += v * proportion1;
537        } else {
538            let v_norm = v.normalize();
539            p0 -= v_norm.clone() * dist0;
540            p1 += v_norm * dist1;
541        }
542
543        self.points[segment_id] = p0;
544        self.points[segment_id + 1] = p1;
545
546        if self.is_closed() {
547            let len = self.points.len();
548            if segment_id == 0 {
549                let first = self.points[0].clone();
550                self.points[len - 1] = first;
551            } else if segment_id + 1 == len - 1 {
552                let last = self.points[len - 1].clone();
553                self.points[0] = last;
554            }
555        }
556    }

558    /// Extend segment equally on both ends (static utility)
559    pub fn extend_segment_equally_static(
560        segment_start: &mut Point,
561        segment_end: &mut Point,
562        dist: f64,
563        proportion: f64,
564    ) {
565        if dist == 0.0 && proportion == 0.0 {
566            return;
567        }
568
569        let v = segment_end.clone() - segment_start.clone();
570
571        if proportion != 0.0 {
572            *segment_start = segment_start.clone() - (v.clone() * proportion);
573            *segment_end = segment_end.clone() + (v * proportion);
574        } else {
575            let mut v_norm = v;
576            v_norm.normalize_self();
577            *segment_start = segment_start.clone() - (v_norm.clone() * dist);
578            *segment_end = segment_end.clone() + (v_norm * dist);
579        }
580    }

582    /// Extend polyline segment equally
583    pub fn extend_segment_equally(&mut self, segment_id: usize, dist: f64, proportion: f64) {
584        if segment_id >= self.segment_count() {
585            return;
586        }
587
588        // Extract points to avoid borrowing issues
589        let mut start = self.points[segment_id].clone();
590        let mut end = self.points[segment_id + 1].clone();
591        Self::extend_segment_equally_static(&mut start, &mut end, dist, proportion);
592        self.points[segment_id] = start;
593        self.points[segment_id + 1] = end;
594
595        if self.points.len() > 2 && self.is_closed() {
596            let len = self.points.len();
597            if segment_id == 0 {
598                self.points[len - 1] = self.points[0].clone();
599            } else if segment_id + 1 == len - 1 {
600                self.points[0] = self.points[len - 1].clone();
601            }
602        }
603    }

612    /// Check if polyline is clockwise oriented
613    pub fn is_clockwise(&self, _plane: &Plane) -> bool {
614        if self.points.len() < 3 {
615            return false;
616        }
617
618        let mut sum = 0.0;
619        let n = if self.is_closed() {
620            self.points.len() - 1
621        } else {
622            self.points.len()
623        };
624
625        for i in 0..n {
626            let current = &self.points[i];
627            let next = &self.points[(i + 1) % n];
628            sum += (next.x() - current.x()) * (next.y() + current.y());
629        }
630
631        sum > 0.0
632    }

639    /// Get convex/concave corners of polyline
640    pub fn get_convex_corners(&self) -> Vec<bool> {
641        if self.points.len() < 3 {
642            return Vec::new();
643        }
644
645        let closed = self.is_closed();
646        let normal = self.average_normal();
647        let n = if closed {
648            self.points.len() - 1
649        } else {
650            self.points.len()
651        };
652        let mut convex_corners = Vec::with_capacity(n);
653
654        for current in 0..n {
655            let prev = if current == 0 { n - 1 } else { current - 1 };
656            let next = if current == n - 1 { 0 } else { current + 1 };
657
658            let mut dir0 = self.points[current].clone() - self.points[prev].clone();
659            dir0.normalize_self();
660
661            let mut dir1 = self.points[next].clone() - self.points[current].clone();
662            dir1.normalize_self();
663
664            let mut cross = dir0.cross(&dir1);
665            cross.normalize_self();
666
667            let dot = cross.dot(&normal);
668            let is_convex = dot >= 0.0;
669            convex_corners.push(is_convex);
670        }
671
672        convex_corners
673    }
674
675    /// Interpolate between two polylines
676    pub fn tween_two_polylines(
677        polyline0: &Polyline,
678        polyline1: &Polyline,
679        weight: f64,
680    ) -> Polyline {
681        if polyline0.points.len() != polyline1.points.len() {
682            return polyline0.clone();
683        }
684
685        let mut result = Polyline::default();
686        result.points.reserve(polyline0.points.len());
687
688        for i in 0..polyline0.points.len() {
689            let diff = polyline1.points[i].clone() - polyline0.points[i].clone();
690            let interpolated = polyline0.points[i].clone() + (diff * weight);
691            result.points.push(interpolated);
692        }
693
694        result
695    }
696
697    /// Calculate average normal from polyline points
698    fn average_normal(&self) -> Vector {
699        let len = self.points.len();
700        if len < 3 {
701            return Vector::new(0.0, 0.0, 1.0);
702        }
703
704        let closed = self.is_closed();
705        let n = if closed && len > 1 { len - 1 } else { len };
706
707        let mut average_normal = Vector::new(0.0, 0.0, 0.0);
708
709        for i in 0..n {
710            let prev = if i == 0 { n - 1 } else { i - 1 };
711            let next = (i + 1) % n;
712
713            let v1 = self.points[prev].clone() - self.points[i].clone();
714            let v2 = self.points[i].clone() - self.points[next].clone();
715            let cross = v1.cross(&v2);
716            average_normal += &cross;
717        }
718
719        average_normal.normalize_self();
720        average_normal
721    }
722}

731        for p in &mut self.points {

739        );
