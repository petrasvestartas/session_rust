36// Lightweight AABB for arena nodes (6 doubles, no axes)
37#[derive(Clone, Copy, Default, Debug)]
38struct BvhAABB {
39    cx: f64,
40    cy: f64,
41    cz: f64,
42    hx: f64,
43    hy: f64,
44    hz: f64,
45}
46
47impl BvhAABB {
48    #[inline(always)]
49    fn from_bbox(b: &BoundingBox) -> Self {
50        BvhAABB {
51            cx: b.center.x(),
52            cy: b.center.y(),
53            cz: b.center.z(),
54            hx: b.half_size.x(),
55            hy: b.half_size.y(),
56            hz: b.half_size.z(),
57        }
58    }
59
60    #[inline(always)]
61    fn merge(a: BvhAABB, b: BvhAABB) -> BvhAABB {
62        let min_x = (a.cx - a.hx).min(b.cx - b.hx);
63        let min_y = (a.cy - a.hy).min(b.cy - b.hy);
64        let min_z = (a.cz - a.hz).min(b.cz - b.hz);
65        let max_x = (a.cx + a.hx).max(b.cx + b.hx);
66        let max_y = (a.cy + a.hy).max(b.cy + b.hy);
67        let max_z = (a.cz + a.hz).max(b.cz + b.hz);
68        BvhAABB {
69            cx: (min_x + max_x) * 0.5,
70            cy: (min_y + max_y) * 0.5,
71            cz: (min_z + max_z) * 0.5,
72            hx: (max_x - min_x) * 0.5,
73            hy: (max_y - min_y) * 0.5,
74            hz: (max_z - min_z) * 0.5,
75        }
76    }
77
78    #[inline(always)]
79    fn intersects(&self, other: &BvhAABB) -> bool {
80        let min1_x = self.cx - self.hx;
81        let max1_x = self.cx + self.hx;
82        let min1_y = self.cy - self.hy;
83        let max1_y = self.cy + self.hy;
84        let min1_z = self.cz - self.hz;
85        let max1_z = self.cz + self.hz;
86
87        let min2_x = other.cx - other.hx;
88        let max2_x = other.cx + other.hx;
89        let min2_y = other.cy - other.hy;
90        let max2_y = other.cy + other.hy;
91        let min2_z = other.cz - other.hz;
92        let max2_z = other.cz + other.hz;
93
94        min1_x <= max2_x
95            && max1_x >= min2_x
96            && min1_y <= max2_y
97            && max1_y >= min2_y
98            && min1_z <= max2_z
99            && max1_z >= min2_z
100    }
101}

151    /// Compute world size from bounding boxes
152    pub fn compute_world_size(bounding_boxes: &[BoundingBox]) -> f64 {
153        if bounding_boxes.is_empty() {
154            return 1000.0;
155        }
156
157        let mut max_extent = 0.0f64;
158        for bbox in bounding_boxes {
159            // Find maximum absolute coordinate in any dimension
160            let x_extent = (bbox.center.x() + bbox.half_size.x())
161                .abs()
162                .max((bbox.center.x() - bbox.half_size.x()).abs());
163            let y_extent = (bbox.center.y() + bbox.half_size.y())
164                .abs()
165                .max((bbox.center.y() - bbox.half_size.y()).abs());
166            let z_extent = (bbox.center.z() + bbox.half_size.z())
167                .abs()
168                .max((bbox.center.z() - bbox.half_size.z()).abs());
169
170            max_extent = max_extent.max(x_extent).max(y_extent).max(z_extent);
171        }
172
173        // World size should be at least 2x the maximum extent, plus padding
174        (max_extent * 2.2).max(10.0)
175    }

209    pub fn build(&mut self, bounding_boxes: &[BoundingBox]) {
210        if bounding_boxes.is_empty() {
211            self.root = None;
212            self.arena.clear();
213            self.arena_root = -1;
214            return;
215        }
216
217        // Create list of objects with their Morton codes (no bbox copies needed later)
218        let mut objects: Vec<ObjectInfo> = bounding_boxes
219            .iter()
220            .enumerate()
221            .map(|(i, bbox)| {
222                let morton_code = calculate_morton_code(
223                    bbox.center.x(),
224                    bbox.center.y(),
225                    bbox.center.z(),
226                    self.world_size,
227                );
228                ObjectInfo { id: i, morton_code }
229            })
230            .collect();
231
232        // Radix sort 30-bit Morton codes: 3 passes of 10 bits (RADIX = 1024)
233        {
234            const RADIX: usize = 1024;
235            const PASSES: usize = 3;
236            let mut tmp: Vec<ObjectInfo> = vec![objects[0].clone(); objects.len()];
237            for pass in 0..PASSES {
238                let shift = (pass * 10) as u32;
239                let mut count = [0usize; RADIX];
240                for e in objects.iter() {
241                    let b = ((e.morton_code >> shift) & ((RADIX as u32) - 1)) as usize;
242                    count[b] += 1;
243                }
244                let mut sum = 0usize;
245                for c in count.iter_mut() {
246                    let old = *c;
247                    *c = sum;
248                    sum += old;
249                }
250                for e in objects.iter() {
251                    let b = ((e.morton_code >> shift) & ((RADIX as u32) - 1)) as usize;
252                    tmp[count[b]] = ObjectInfo {
253                        id: e.id,
254                        morton_code: e.morton_code,
255                    };
256                    count[b] += 1;
257                }
258                std::mem::swap(&mut objects, &mut tmp);
259            }
260        }
261
262        // LBVH (Karras 2012) construction in O(N) after sort
263        let n = objects.len();
264        if n == 1 {
265            // Single leaf - build arena only
266            let id = objects[0].id;
267            let aabb = BvhAABB::from_bbox(&bounding_boxes[id]);
268
269            self.arena.clear();
270            self.arena.push(FlatNode {
271                left: -1,
272                right: -1,
273                object_id: id as i32,
274                aabb,
275            });
276            self.arena_root = 0;
277            self.root = None;
278            return;
279        }
280
281        // Codes array for prefix computations
282        let codes: Vec<u32> = objects.iter().map(|o| o.morton_code).collect();
283
284        #[inline]
285        fn clz32(x: u32) -> i32 {
286            if x == 0 {
287                32
288            } else {
289                x.leading_zeros() as i32
290            }
291        }
292
293        let common_prefix = |i: i32, j: i32| -> i32 {
294            if j < 0 || j >= n as i32 {
295                return -1;
296            }
297            let ci = codes[i as usize];
298            let cj = codes[j as usize];
299            if ci != cj {
300                return clz32(ci ^ cj);
301            }
302            let di = i as u32;
303            let dj = j as u32;
304            32 + clz32(di ^ dj)
305        };
306
307        let determine_range = |i: i32| -> (i32, i32) {
308            let d = if common_prefix(i, i + 1) - common_prefix(i, i - 1) > 0 {
309                1
310            } else {
311                -1
312            };
313            let delta_min = common_prefix(i, i - d);
314            let mut l = 1i32;
315            while common_prefix(i, i + l * d) > delta_min {
316                l <<= 1;
317            }
318            let mut bound = 0i32;
319            let mut t = l >> 1;
320            while t > 0 {
321                if common_prefix(i, i + (bound + t) * d) > delta_min {
322                    bound += t;
323                }
324                t >>= 1;
325            }
326            let j = i + bound * d;
327            (i.min(j), i.max(j))
328        };
329
330        let find_split = |first: i32, last: i32| -> i32 {
331            let common = common_prefix(first, last);
332            let mut split = first;
333            let mut step = last - first;
334            loop {
335                step = (step + 1) >> 1;
336                let new_split = split + step;
337                if new_split < last {
338                    let split_prefix = common_prefix(first, new_split);
339                    if split_prefix > common {
340                        split = new_split;
341                    }
342                }
343                if step <= 1 {
344                    break;
345                }
346            }
347            split
348        };
349
350        // Temporary nodes with child indices
351        #[derive(Clone)]
352        enum TempChild {
353            Leaf(usize),
354            Internal(usize),
355        }
356        #[derive(Clone)]
357        struct TempNode {
358            left: Option<TempChild>,
359            right: Option<TempChild>,
360            object_id: i32,
361            aabb: BvhAABB,
362        }
363
364        // Allocate leaves
365        let mut leaves: Vec<TempNode> = Vec::with_capacity(n);
366        for obj in objects.iter() {
367            let id = obj.id;
368            let aabb = BvhAABB::from_bbox(&bounding_boxes[id]);
369            leaves.push(TempNode {
370                left: None,
371                right: None,
372                object_id: id as i32,
373                aabb,
374            });
375        }
376
377        // Allocate internals
378        let mut internals: Vec<TempNode> = Vec::with_capacity(n - 1);
379        for _ in 0..(n - 1) {
380            internals.push(TempNode {
381                left: None,
382                right: None,
383                object_id: -1,
384                aabb: BvhAABB::default(),
385            });
386        }
387
388        // Build topology
389        let mut has_parent: Vec<bool> = vec![false; n - 1];
390        for i in 0..(n as i32 - 1) {
391            let (first, last) = determine_range(i);
392            let split = find_split(first, last);
393            if split == first {
394                internals[i as usize].left = Some(TempChild::Leaf(split as usize));
395            } else {
396                internals[i as usize].left = Some(TempChild::Internal(split as usize));
397                has_parent[split as usize] = true;
398            }
399            if split + 1 == last {
400                internals[i as usize].right = Some(TempChild::Leaf((split + 1) as usize));
401            } else {
402                internals[i as usize].right = Some(TempChild::Internal((split + 1) as usize));
403                has_parent[(split + 1) as usize] = true;
404            }
405        }
406
407        // Find root internal index
408        let mut root_idx: usize = 0;
409        for (idx, hp) in has_parent.iter().enumerate() {
410            if !*hp {
411                root_idx = idx;
412                break;
413            }
414        }
415
416        // Post-order compute internal AABBs
417        fn compute_internal(
418            idx: usize,
419            internals: &mut [TempNode],
420            leaves: &[TempNode],
421        ) -> BvhAABB {
422            let (left, right) = (
423                internals[idx].left.clone().expect("left child"),
424                internals[idx].right.clone().expect("right child"),
425            );
426            let a = match left {
427                TempChild::Leaf(li) => leaves[li].aabb,
428                TempChild::Internal(ii) => compute_internal(ii, internals, leaves),
429            };
430            let b = match right {
431                TempChild::Leaf(li) => leaves[li].aabb,
432                TempChild::Internal(ii) => compute_internal(ii, internals, leaves),
433            };
434            let merged = BvhAABB::merge(a, b);
435            internals[idx].aabb = merged;
436            merged
437        }
438        compute_internal(root_idx, &mut internals, &leaves);
439
440        // Build flat arena for fast queries (cache-friendly, no pointers)
441        self.arena.clear();
442        self.arena.reserve(n + (n - 1)); // Reserve for all nodes
443
444        // Helper to convert TempNode to FlatNode and add to arena
445        fn build_arena_node(
446            node_ref: TempChild,
447            internals: &[TempNode],
448            leaves: &[TempNode],
449            arena: &mut Vec<FlatNode>,
450        ) -> i32 {
451            match node_ref {
452                TempChild::Leaf(li) => {
453                    let idx = arena.len() as i32;
454                    arena.push(FlatNode {
455                        left: -1,
456                        right: -1,
457                        object_id: leaves[li].object_id,
458                        aabb: leaves[li].aabb,
459                    });
460                    idx
461                }
462                TempChild::Internal(ii) => {
463                    let idx = arena.len() as i32;
464                    // Reserve slot
465                    arena.push(FlatNode {
466                        left: -1,
467                        right: -1,
468                        object_id: -1,
469                        aabb: internals[ii].aabb,
470                    });
471                    // Build children
472                    let left_idx = build_arena_node(
473                        internals[ii].left.clone().expect("left child"),
474                        internals,
475                        leaves,
476                        arena,
477                    );
478                    let right_idx = build_arena_node(
479                        internals[ii].right.clone().expect("right child"),
480                        internals,
481                        leaves,
482                        arena,
483                    );
484                    // Update slot
485                    arena[idx as usize].left = left_idx;
486                    arena[idx as usize].right = right_idx;
487                    idx
488                }
489            }
490        }
491
492        self.arena_root = build_arena_node(
493            TempChild::Internal(root_idx),
494            &internals,
495            &leaves,
496            &mut self.arena,
497        );
498
499        // Leave self.root as None - arena is used for all queries now
500        self.root = None;
501    }

503    pub fn merge_aabb(&self, aabb1: &BoundingBox, aabb2: &BoundingBox) -> BoundingBox {
504        // Calculate min and max corners
505        let min_x =
506            (aabb1.center.x() - aabb1.half_size.x()).min(aabb2.center.x() - aabb2.half_size.x());
507        let min_y =
508            (aabb1.center.y() - aabb1.half_size.y()).min(aabb2.center.y() - aabb2.half_size.y());
509        let min_z =
510            (aabb1.center.z() - aabb1.half_size.z()).min(aabb2.center.z() - aabb2.half_size.z());
511
512        let max_x =
513            (aabb1.center.x() + aabb1.half_size.x()).max(aabb2.center.x() + aabb2.half_size.x());
514        let max_y =
515            (aabb1.center.y() + aabb1.half_size.y()).max(aabb2.center.y() + aabb2.half_size.y());
516        let max_z =
517            (aabb1.center.z() + aabb1.half_size.z()).max(aabb2.center.z() + aabb2.half_size.z());
518
519        // Calculate new center and half_size
520        let center = Point::new(
521            (min_x + max_x) / 2.0,
522            (min_y + max_y) / 2.0,
523            (min_z + max_z) / 2.0,
524        );
525        let half_size = Vector::new(
526            (max_x - min_x) / 2.0,
527            (max_y - min_y) / 2.0,
528            (max_z - min_z) / 2.0,
529        );
530
531        BoundingBox::new(
532            center,
533            Vector::new(1.0, 0.0, 0.0),
534            Vector::new(0.0, 1.0, 0.0),
535            Vector::new(0.0, 0.0, 1.0),
536            half_size,
537        )
538    }

540    pub fn find_collisions(
541        &self,
542        object_id: usize,
543        query_bbox: &BoundingBox,
544        bounding_boxes: &[BoundingBox],
545    ) -> (Vec<usize>, i32) {
546        let mut collisions = Vec::new();
547        let mut check_count = 0;
548
549        // Use arena for fast traversal
550        if self.arena_root < 0 || self.arena.is_empty() {
551            return (collisions, check_count);
552        }
553
554        let query_aabb = BvhAABB::from_bbox(query_bbox);
555        let mut stack: Vec<i32> = Vec::with_capacity(64);
556        stack.push(self.arena_root);
557
558        while let Some(node_idx) = stack.pop() {
559            check_count += 1;
560            let node = &self.arena[node_idx as usize];
561
562            // Early exit if query doesn't intersect this node's AABB
563            if !query_aabb.intersects(&node.aabb) {
564                continue;
565            }
566
567            // If leaf node, check for collision
568            if node.object_id >= 0 {
569                let node_object_id = node.object_id as usize;
570                // Don't check collision with self
571                if node_object_id != object_id
572                    && node_object_id < bounding_boxes.len()
573                    && self.aabb_intersect(query_bbox, &bounding_boxes[node_object_id])
574                {
575                    collisions.push(node_object_id);
576                }
577                continue;
578            }
579
580            // Internal node: push children
581            if node.left >= 0 {
582                stack.push(node.left);
583            }
584            if node.right >= 0 {
585                stack.push(node.right);
586            }
587        }
588
589        (collisions, check_count)
590    }

592    pub fn aabb_intersect(&self, aabb1: &BoundingBox, aabb2: &BoundingBox) -> bool {
593        // Calculate min/max for both boxes
594        let min1_x = aabb1.center.x() - aabb1.half_size.x();
595        let max1_x = aabb1.center.x() + aabb1.half_size.x();
596        let min1_y = aabb1.center.y() - aabb1.half_size.y();
597        let max1_y = aabb1.center.y() + aabb1.half_size.y();
598        let min1_z = aabb1.center.z() - aabb1.half_size.z();
599        let max1_z = aabb1.center.z() + aabb1.half_size.z();
600
601        let min2_x = aabb2.center.x() - aabb2.half_size.x();
602        let max2_x = aabb2.center.x() + aabb2.half_size.x();
603        let min2_y = aabb2.center.y() - aabb2.half_size.y();
604        let max2_y = aabb2.center.y() + aabb2.half_size.y();
605        let min2_z = aabb2.center.z() - aabb2.half_size.z();
606        let max2_z = aabb2.center.z() + aabb2.half_size.z();
607
608        // Check for overlap on all three axes
609        min1_x <= max2_x
610            && max1_x >= min2_x
611            && min1_y <= max2_y
612            && max1_y >= min2_y
613            && min1_z <= max2_z
614            && max1_z >= min2_z
615    }

617    pub fn check_all_collisions(
618        &self,
619        bounding_boxes: &[BoundingBox],
620    ) -> (Vec<(usize, usize)>, Vec<usize>, i32) {
621        let mut all_collisions: Vec<(usize, usize)> = Vec::new();
622        let mut total_checks: i32 = 0;
623
624        // Early out if arena is empty
625        if self.arena_root < 0 || self.arena.is_empty() {
626            return (all_collisions, Vec::new(), total_checks);
627        }
628
629        // Track which object indices participate in any collision
630        let mut visited: Vec<bool> = vec![false; bounding_boxes.len()];
631
632        // Stack of node index pairs for pairwise BVH traversal (cache-friendly)
633        let mut stack: Vec<(i32, i32)> = Vec::with_capacity(256);
634        stack.push((self.arena_root, self.arena_root));
635
636        while let Some((a_idx, b_idx)) = stack.pop() {
637            let a = &self.arena[a_idx as usize];
638            let b = &self.arena[b_idx as usize];
639
640            // AABB overlap test (inline for speed)
641            if !a.aabb.intersects(&b.aabb) {
642                continue;
643            }
644
645            // Count only when node AABBs overlap (matches C++ metric)
646            total_checks += 1;
647
648            let a_leaf = a.object_id >= 0;
649            let b_leaf = b.object_id >= 0;
650
651            if a_leaf && b_leaf {
652                let i = a.object_id as usize;
653                let j = b.object_id as usize;
654                if i < j && i < visited.len() && j < visited.len() {
655                    all_collisions.push((i, j));
656                    visited[i] = true;
657                    visited[j] = true;
658                }
659                continue;
660            }
661
662            // Expand children (index-based, no pointer chasing)
663            if a_idx == b_idx {
664                // Same node: expand unique child pairs without symmetry duplicates
665                if a.left >= 0 {
666                    stack.push((a.left, a.left));
667                    if a.right >= 0 {
668                        stack.push((a.left, a.right));
669                        stack.push((a.right, a.right));
670                    }
671                }
672                continue;
673            }
674
675            if !a_leaf && !b_leaf {
676                // Both internal
677                if a.left >= 0 && b.left >= 0 {
678                    stack.push((a.left, b.left));
679                }
680                if a.left >= 0 && b.right >= 0 {
681                    stack.push((a.left, b.right));
682                }
683                if a.right >= 0 && b.left >= 0 {
684                    stack.push((a.right, b.left));
685                }
686                if a.right >= 0 && b.right >= 0 {
687                    stack.push((a.right, b.right));
688                }
689            } else if a_leaf && !b_leaf {
690                // a is leaf, b is internal
691                if b.left >= 0 {
692                    stack.push((a_idx, b.left));
693                }
694                if b.right >= 0 {
695                    stack.push((a_idx, b.right));
696                }
697            } else if !a_leaf && b_leaf {
698                // a is internal, b is leaf
699                if a.left >= 0 {
700                    stack.push((a.left, b_idx));
701                }
702                if a.right >= 0 {
703                    stack.push((a.right, b_idx));
704                }
705            }
706        }
707
708        let mut colliding_indices: Vec<usize> = visited
709            .iter()
710            .enumerate()
711            .filter_map(|(idx, v)| if *v { Some(idx) } else { None })
712            .collect();
713        colliding_indices.sort_unstable();
714
715        (all_collisions, colliding_indices, total_checks)
716    }

718    /// Check for all collisions and return GUID pairs directly
719    /// Uses the internally stored object_guids from build_with_guids
720    pub fn check_all_collisions_guids(
721        &self,
722        bounding_boxes: &[BoundingBox],
723    ) -> Vec<(String, String)> {
724        let (collision_pairs, _, _) = self.check_all_collisions(bounding_boxes);
725
726        // Convert indices to GUIDs
727        collision_pairs
728            .iter()
729            .filter_map(|(i, j)| {
730                if *i < self.object_guids.len() && *j < self.object_guids.len() {
731                    Some((self.object_guids[*i].clone(), self.object_guids[*j].clone()))
732                } else {
733                    None
734                }
735            })
736            .collect()
737    }

739    #[inline(always)]
740    fn ray_bvhaabb_intersect(
741        origin: &Point,
742        direction: &Vector,
743        aabb: &BvhAABB,
744    ) -> Option<(f64, f64)> {
745        let min_x = aabb.cx - aabb.hx;
746        let max_x = aabb.cx + aabb.hx;
747        let min_y = aabb.cy - aabb.hy;
748        let max_y = aabb.cy + aabb.hy;
749        let min_z = aabb.cz - aabb.hz;
750        let max_z = aabb.cz + aabb.hz;
751
752        let invx = if direction.x() == 0.0 {
753            f64::INFINITY
754        } else {
755            1.0 / direction.x()
756        };
757        let invy = if direction.y() == 0.0 {
758            f64::INFINITY
759        } else {
760            1.0 / direction.y()
761        };
762        let invz = if direction.z() == 0.0 {
763            f64::INFINITY
764        } else {
765            1.0 / direction.z()
766        };
767
768        let tx1 = (min_x - origin.x()) * invx;
769        let tx2 = (max_x - origin.x()) * invx;
770        let mut tmin = tx1.min(tx2);
771        let mut tmax = tx1.max(tx2);
772
773        let ty1 = (min_y - origin.y()) * invy;
774        let ty2 = (max_y - origin.y()) * invy;
775        tmin = tmin.max(ty1.min(ty2));
776        tmax = tmax.min(ty1.max(ty2));
777
778        let tz1 = (min_z - origin.z()) * invz;
779        let tz2 = (max_z - origin.z()) * invz;
780        tmin = tmin.max(tz1.min(tz2));
781        tmax = tmax.min(tz1.max(tz2));
782
783        if tmax >= tmin {
784            Some((tmin, tmax))
785        } else {
786            None
787        }
788    }

790    pub fn ray_cast(
791        &self,
792        origin: &Point,
793        direction: &Vector,
794        candidate_leaf_ids: &mut Vec<usize>,
795        _find_all: bool,
796    ) -> bool {
797        candidate_leaf_ids.clear();
798
799        // Use arena for fast index-based traversal
800        if self.arena_root < 0 || self.arena.is_empty() {
801            return false;
802        }
803
804        let root = &self.arena[self.arena_root as usize];
805        if let Some((_tmin, tmax)) = Self::ray_bvhaabb_intersect(origin, direction, &root.aabb) {
806            if tmax < 0.0 {
807                return false;
808            }
809        } else {
810            return false;
811        }
812
813        let mut stack: Vec<i32> = Vec::with_capacity(64);
814        stack.push(self.arena_root);
815
816        while let Some(node_idx) = stack.pop() {
817            let node = &self.arena[node_idx as usize];
818
819            if let Some((_tmin, tmax)) = Self::ray_bvhaabb_intersect(origin, direction, &node.aabb)
820            {
821                if tmax < 0.0 {
822                    continue;
823                }
824            } else {
825                continue;
826            }
827
828            if node.object_id >= 0 {
829                // Leaf node
830                candidate_leaf_ids.push(node.object_id as usize);
831                continue;
832            }
833
834            // Internal node: push children
835            if node.left >= 0 {
836                stack.push(node.left);
837            }
838            if node.right >= 0 {
839                stack.push(node.right);
840            }
841        }
842
843        !candidate_leaf_ids.is_empty()
844    }
845}

847// Morton code functions
848pub fn expand_bits(v: u32) -> u32 {
849    let mut v = v;
850    v = (v.wrapping_mul(0x00010001)) & 0xFF0000FF;
851    v = (v.wrapping_mul(0x00000101)) & 0x0F00F00F;
852    v = (v.wrapping_mul(0x00000011)) & 0xC30C30C3;
853    v = (v.wrapping_mul(0x00000005)) & 0x49249249;
854    v
855}

857pub fn calculate_morton_code(x: f64, y: f64, z: f64, world_size: f64) -> u32 {
858    // Normalize coordinates to [0,1] range
859    let nx = (x + world_size / 2.0) / world_size;
860    let ny = (y + world_size / 2.0) / world_size;
861    let nz = (z + world_size / 2.0) / world_size;
862
863    // Clamp to [0,1]
864    let nx = nx.clamp(0.0, 1.0);
865    let ny = ny.clamp(0.0, 1.0);
866    let nz = nz.clamp(0.0, 1.0);
867
868    // Scale to [0, 1023] for 10-bit encoding
869    let ix = ((nx * 1023.0) as u32).min(1023);
870    let iy = ((ny * 1023.0) as u32).min(1023);
871    let iz = ((nz * 1023.0) as u32).min(1023);
872
873    // Expand bits and interleave
874    let xx = expand_bits(ix);
875    let yy = expand_bits(iy);
876    let zz = expand_bits(iz);
877
878    xx | (yy << 1) | (zz << 2)
879}
